The Rust SDK is async, type-safe, and built on `tokio` + `reqwest` + `serde`. All protocol types derive `Serialize + Deserialize + Clone + Debug`.

<Section id="install" title="Install">

```toml
[dependencies]
wechatbot = "0.1"
tokio = { version = "1", features = ["full"] }
```

</Section>

<Section id="quick-start" title="Quick start">

```rust
use wechatbot::{WeChatBot, BotOptions};

#[tokio::main]
async fn main() {
    let bot = WeChatBot::new(BotOptions::default());
    let creds = bot.login(false).await.unwrap();
    println!("Logged in: {}", creds.account_id);

    bot.on_message(Box::new(|msg| {
        println!("{}: {}", msg.user_id, msg.text);
    })).await;

    bot.run().await.unwrap();
}
```

</Section>

<Section id="configuration" title="Configuration">

```rust
let bot = WeChatBot::new(BotOptions {
    base_url: None,     // default: ilinkai.weixin.qq.com
    cred_path: None,    // default: ~/.wechatbot/credentials.json
    on_qr_url: Some(Box::new(|url| {
        println!("Scan: {}", url);
    })),
    on_error: Some(Box::new(|err| {
        eprintln!("Error: {}", err);
    })),
});
```

<Section id="messages" title="Message handling">

```rust
bot.on_message(Box::new(|msg| {
    match msg.content_type {
        ContentType::Text  => println!("Text: {}", msg.text),
        ContentType::Image => {
            for img in &msg.images {
                println!("Image: {:?}", img.url);
            }
        }
        ContentType::Voice => {
            for v in &msg.voices {
                println!("Voice: {:?} ({}ms)", v.text, v.duration_ms.unwrap_or(0));
            }
        }
        ContentType::File => {
            for f in &msg.files { println!("File: {:?}", f.file_name); }
        }
        ContentType::Video => println!("Video received"),
    }
})).await;
```

</Section>

<Section id="sending" title="Sending messages">

```rust
// Reply
bot.reply(&msg, "Echo: hello").await?;

// Send (needs prior context_token)
bot.send(user_id, "Hello").await?;

// Typing
bot.send_typing(user_id).await?;
```

</Section>

<Section id="errors" title="Error handling">

```rust
use wechatbot::WeChatBotError;

match result {
    Err(WeChatBotError::Api { errcode: -14, .. }) => {
        // session expired — handled automatically
    }
    Err(WeChatBotError::NoContext(uid)) => {
        // no context_token yet
    }
    Err(WeChatBotError::Transport(e)) => {
        // network error
    }
    _ => {}
}
```

</Section>

<Section id="crypto" title="AES-128-ECB crypto">

```rust
use wechatbot::{generate_aes_key, encrypt_aes_ecb, decrypt_aes_ecb, decode_aes_key};

let key = generate_aes_key();
let ct = encrypt_aes_ecb(b"Hello", &key);
let pt = decrypt_aes_ecb(&ct, &key)?;

// Decode protocol key (all 3 formats)
let k = decode_aes_key("ABEiM0RVZneImaq7zN3u/w==")?;     // base64(raw)
let k = decode_aes_key("00112233445566778899aabbccddeeff")?; // hex
```

</Section>

<Section id="structure" title="Project structure">

```
rust/src/
├── lib.rs           ← Public re-exports
├── types.rs         ← All types (serde)
├── error.rs         ← Error hierarchy (thiserror)
├── protocol.rs      ← iLink API calls (reqwest)
├── crypto.rs        ← AES-128-ECB + tests
└── bot.rs           ← WeChatBot client
```
