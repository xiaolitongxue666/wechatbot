The Go SDK is clean, idiomatic, and goroutine-safe. **stdlib only**, no CGO. Uses `sync.Map` for concurrent context_token access.

<Section id="install" title="Install">

```bash
go get github.com/anthropic/wechatbot-go
```

</Section>

<Section id="quick-start" title="Quick start">

```go
package main

import (
    "context"
    "fmt"
    wechatbot "github.com/anthropic/wechatbot-go"
)

func main() {
    ctx := context.Background()
    bot := wechatbot.New()

    creds, _ := bot.Login(ctx, false)
    fmt.Printf("Logged in: %s\n", creds.AccountID)

    bot.OnMessage(func(msg *wechatbot.IncomingMessage) {
        bot.SendTyping(ctx, msg.UserID)
        bot.Reply(ctx, msg, fmt.Sprintf("Echo: %s", msg.Text))
    })

    bot.Run(ctx)
}
```

</Section>

<Section id="messages" title="Message handling">

```go
bot.OnMessage(func(msg *wechatbot.IncomingMessage) {
    fmt.Printf("[%s] %s: %s\n", msg.Type, msg.UserID, msg.Text)

    for _, img := range msg.Images {
        fmt.Printf("  Image: %s\n", img.URL)
    }
    for _, voice := range msg.Voices {
        fmt.Printf("  Voice: %s (%dms)\n", voice.Text, voice.DurationMs)
    }
    for _, file := range msg.Files {
        fmt.Printf("  File: %s (%d bytes)\n", file.FileName, file.Size)
    }
    if msg.QuotedMessage != nil {
        fmt.Printf("  Quoted: %s\n", msg.QuotedMessage.Title)
    }
})
```

</Section>

<Section id="sending" title="Sending messages">

```go
// Reply (auto context_token)
bot.Reply(ctx, msg, "Hello back!")

// Send to user (needs prior context)
bot.Send(ctx, userID, "Proactive message")

// Typing
bot.SendTyping(ctx, userID)
bot.StopTyping(ctx, userID)
```

</Section>

<Section id="lifecycle" title="Lifecycle">

```go
// Start (blocks until Stop or context cancel)
err := bot.Run(ctx)

// Stop gracefully
bot.Stop()
```

Session recovery is automatic — on `-14` error, the bot clears state, deletes credentials, and re-runs QR login.

</Section>

<Section id="types" title="Key types">

```go
type IncomingMessage struct {
    UserID        string
    Text          string
    Type          ContentType  // "text", "image", "voice", "file", "video"
    Timestamp     time.Time
    Images        []ImageContent
    Voices        []VoiceContent
    Files         []FileContent
    Videos        []VideoContent
    QuotedMessage *QuotedMessage
}
```

</Section>

<Section id="structure" title="Project structure">

```
golang/
├── types.go                 ← All public types
├── bot.go                   ← Bot client (login, run, reply, send)
├── internal/
│   ├── protocol/api.go      ← Raw iLink HTTP calls
│   ├── auth/login.go        ← QR login + credential persistence
│   └── crypto/
│       ├── aes.go           ← AES-128-ECB encrypt/decrypt
│       └── aes_test.go      ← Crypto tests
└── examples/
    └── echo-bot/main.go
```
